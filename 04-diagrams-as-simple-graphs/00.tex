\usetikzlibrary{
  positioning,
  shapes.misc,
  calc,
  graphs,
  arrows.meta,
}

\tikzset{
  node distance= 5mm and 5mm,
  text height=1.5ex,
  text depth=.25ex,
  >={Stealth[round]},
  every new ->/.style={
    shorten >=1pt,
    thick, 
    black!50, 
    text=black
  },
  graphs/every graph/.style={edges=rounded corners},
  node/.style={
    rectangle,
    minimum size=6mm,
    very thick,
  },
  nonterminal/.style={
    node,
    draw=red!50!black!50,
    top color=white,
    bottom color=red!50!black!20,
    font=\itshape,
  },
  terminal/.style={
    node,
    rounded corners=3mm,
    draw=black!50,
    top color=white,
    bottom color=black!20,
    font=\ttfamily,
  },
}
